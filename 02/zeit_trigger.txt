create or replace trigger Date_Exporttrigger
after insert or update on Lieferungen
for each row
declare 
    datum   date;
    jahr    number(4);
    monat   number(2);
    tag     number(2);
begin
    select Lieferdatum into datum from lieferungen;
    select extract(year from Lieferdatum) into jahr from Lieferungen;
    select extract(month from Lieferdatum) into monat from Lieferungen;
    select extract(day from Lieferdatum) into tag from Lieferungen;
    Insert into Zeit values(datum, jahr, monat, tag );
end;
